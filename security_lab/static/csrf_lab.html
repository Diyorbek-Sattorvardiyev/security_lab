XAVFSIZ (Token tekshiriladi) 

@app.post("/profile/update")
async def update(data: dict, req: Request):
  token = data.get("csrf_token")
  sid = req.cookies.get("session_id")
  if not validate_csrf_token(sid, token):
    raise HTTPException(403, "CSRF!")
  db.update(data)
  return {"ok": True}
